import pandas as pd

csv_dataset_names = ["data_quiz/wdi_agedist_1980.csv", "data_quiz/wdi_agedist_2000.csv", "data_quiz/wdi_agedist_2020.csv"]

list_newnames = [name.replace(".csv", ".xlsx").replace("data_quiz/", "data_excel/") for name in csv_dataset_names]

index = 0

for dataset_name in csv_dataset_names:
    dataset = pd.read_csv(dataset_name)
    
    filename = list_newnames[index]
    
    dataset.to_excel(filename, index=False)
    
    index += 1

csv_dataset_names = ["data_quiz/wdi_agedist_1980.csv", "data_quiz/wdi_agedist_2000.csv", "data_quiz/wdi_agedist_2020.csv"]

index = 0

for dataset_name in csv_dataset_names:
    dataset = pd.read_csv(dataset_name)
    
    percent_ages0to14 = dataset["percent_ages0to14"]
    
    percent_ages15andolder = 100 - percent_ages0to14
    
    plt.hist(percent_ages15andolder, bins=20, color='blue', alpha=0.7)
    plt.title(f'Histogram of Percentages Ages 15 and Older ({dataset_name})')
    plt.xlabel('Percentage')
    plt.ylabel('Frequency')
    
    plt.show()
    
    index += 1
